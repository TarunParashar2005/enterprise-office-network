
 Device: ISP-ROUTER (Cisco 2811)
 Role  : Simulated ISP / Internet


enable
configure terminal

interface FastEthernet0/0
 description LINK-TO-EDGE
 ip address 200.0.0.1 255.255.255.0
 no shutdown
exit

interface Loopback0
 description INTERNET-SIMULATION
 ip address 8.8.8.8 255.255.255.255
 no shutdown
exit

! ---------- Return routes to internal networks ----------
ip route 192.168.10.0 255.255.255.0 200.0.0.2
ip route 192.168.12.0 255.255.255.0 200.0.0.2
ip route 192.168.20.0 255.255.255.0 200.0.0.2
